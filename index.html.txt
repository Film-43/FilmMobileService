<script>
    let currentUser = sessionStorage.getItem('FILM_USER_NAME') || "";
    let items = JSON.parse(localStorage.getItem('FILM_DB_FINAL')) || [];
    let currentCategory = 'ทั้งหมด';
    let tempImgBase64 = "";

    function togglePass(id, el) {
        const input = document.getElementById(id);
        input.type = input.type === "password" ? "text" : "password";
        el.innerText = input.type === "password" ? "👁️" : "🔒";
    }
    function switchAuth(target) {
        document.querySelectorAll('.auth-card').forEach(c => c.classList.remove('active'));
        document.getElementById(target + 'Form').classList.add('active');
    }
    function handleRegister() {
        const u = document.getElementById('regUser').value.trim();
        const p = document.getElementById('regPass').value.trim();
        if(!u || !p) return alert("กรุณากรอกข้อมูลให้ครบ");
        localStorage.setItem(`USER_${u}`, p);
        alert("✅ สมัครสมาชิกสำเร็จ!"); switchAuth('login');
    }
    function handleLogin() {
        const u = document.getElementById('userIn').value.trim();
        const p = document.getElementById('passIn').value.trim();
        const storedPass = localStorage.getItem(`USER_${u}`);
        if((u === 'admin' && p === '1234') || (storedPass && storedPass === p)) {
            sessionStorage.setItem('FILM_SESSION', 'active');
            sessionStorage.setItem('FILM_USER_NAME', u);
            currentUser = u;
            document.getElementById('authOverlay').style.display = 'none';
        } else alert("❌ ชื่อผู้ใช้งานหรือรหัสผ่านไม่ถูกต้อง");
    }
    function handleReset() {
        const u = document.getElementById('forgotUser').value.trim();
        const p = document.getElementById('newPassForgot').value.trim();
        if(!localStorage.getItem(`USER_${u}`)) return alert("❌ ไม่พบชื่อผู้ใช้งานนี้");
        localStorage.setItem(`USER_${u}`, p);
        alert("✅ รีเซ็ตรหัสผ่านสำเร็จ!"); switchAuth('login');
    }
    function logout() { sessionStorage.clear(); location.reload(); }

    function openSettingModal() {
        document.getElementById('currUser').value = currentUser || "admin";
        document.getElementById('changePassIn').value = "";
        document.getElementById('settingModal').style.display = 'flex';
    }
    function saveNewPassword() {
        const newP = document.getElementById('changePassIn').value.trim();
        if(!newP) return alert("กรุณากรอกรหัสผ่านใหม่");
        localStorage.setItem(`USER_${currentUser || 'admin'}`, newP);
        alert("✅ เปลี่ยนรหัสผ่านสำเร็จ!"); closeAllModals();
    }

    // ระบบ Validation
    document.querySelectorAll('.validate').forEach(el => {
        el.addEventListener('input', function() { if(this.value.trim() !== "") this.classList.remove('error-flash'); });
    });

    function renderUI() {
        const grid = document.getElementById('productGrid'); grid.innerHTML = "";
        const search = document.getElementById('searchInput').value.toLowerCase();
        items.filter(i => (currentCategory === 'ทั้งหมด' || i.cat === currentCategory) && i.name.toLowerCase().includes(search)).forEach((item, idx) => {
            const card = document.createElement('div'); card.className = 'card';
            card.innerHTML = `
                <div class="p-img-box" onclick="viewProduct(${idx})"><img src="${item.img}"></div>
                <div class="p-info">
                    <div class="p-name" onclick="viewProduct(${idx})">${item.name}</div>
                    <div class="p-price">฿${Number(item.price).toLocaleString()}</div>
                    <div class="p-meta">คงเหลือ: ${item.qty} | ${item.code}</div>
                    <button class="btn-edit-card" onclick="editItem(${idx})">แก้ไข</button>
                    <button class="btn-del-card" onclick="deleteItem(${idx})">ลบรายการ</button>
                </div>`;
            grid.appendChild(card);
        });
    }

    function viewProduct(i) {
        const it = items[i];
        document.getElementById('viewImg').src = it.img;
        document.getElementById('viewName').innerText = it.name;
        document.getElementById('viewPrice').innerText = `฿${Number(it.price).toLocaleString()}`;
        document.getElementById('viewModal').style.display = 'flex';
    }

    // ฟังก์ชันจัดการไฟล์รูปภาพ (บล็อกไฟล์ใหญ่)
    function handleFile(input) {
        const previewBox = document.getElementById('previewBox');
        const fileNameDisplay = document.getElementById('fileNameDisplay');
        const uploadArea = document.getElementById('uploadArea');

        if (input.files && input.files[0]) {
            const file = input.files[0];
            const fileSizeKB = file.size / 1024;
            const limitKB = 500; 

            if (fileSizeKB > limitKB) {
                alert(`❌ ไฟล์ใหญ่เกินไป! (${fileSizeKB.toFixed(2)} KB)\nกรุณาใช้รูปไม่เกิน ${limitKB} KB ครับ`);
                input.value = ""; 
                tempImgBase64 = "";

    function processSave() {
        const fields = ['inCat', 'inName', 'inPrice', 'inQty', 'inCode'];
        let isValid = true;
        fields.forEach(id => {
            const el = document.getElementById(id);
            if(!el.value.trim()) { el.classList.add('error-flash'); isValid = false; }
            else el.classList.remove('error-flash');
        });
        
        if(!tempImgBase64) { 
            document.getElementById('uploadArea').classList.add('error-flash'); 
            isValid = false; 
        }

        if(!isValid) return alert("กรุณากรอกข้อมูลและเลือกรูปภาพให้ถูกต้อง");

        const nameVal = document.getElementById('inName').value.trim();
        const codeVal = document.getElementById('inCode').value.trim();
        const editIdx = document.getElementById('editIdx').value;

        const isDupName = items.some((it, i) => it.name.toLowerCase() === nameVal.toLowerCase() && i.toString() !== editIdx);
        const isDupCode = items.some((it, i) => it.code.toLowerCase() === codeVal.toLowerCase() && i.toString() !== editIdx);

        if(isDupName || isDupCode) {
            alert(`❌ ข้อมูลซ้ำ: ${isDupName ? 'ชื่อรุ่นมีแล้ว' : 'รหัสมีแล้ว'}`);
            if(isDupName) document.getElementById('inName').classList.add('error-flash');
            if(isDupCode) document.getElementById('inCode').classList.add('error-flash');
            return;
        }

        const data = { 
            name: nameVal, 
            price: document.getElementById('inPrice').value, 
            code: codeVal, 
            qty: document.getElementById('inQty').value, 
            cat: document.getElementById('inCat').value, 
            img: tempImgBase64 
        };
        
        let backupItems = [...items];
        if(editIdx === "") items.unshift(data); 
        else items[editIdx] = data;

        try {
            localStorage.setItem('FILM_DB_FINAL', JSON.stringify(items));
            alert("✅ บันทึกรายการสำเร็จ!");
            currentCategory = 'ทั้งหมด';
            document.querySelectorAll('.cat-tab').forEach(t => t.classList.remove('active'));
            document.getElementById('tabAll').classList.add('active');
            closeAllModals();
            renderUI();
        } catch (e) {
            items = backupItems;
            alert("⚠️ พื้นที่จัดเก็บเต็ม! กรุณาลบรายการเก่าออกก่อนเพิ่มใหม่");
        }
    }

    // Fireworks
    const canvas = document.getElementById('fireworksCanvas');
    const ctx = canvas.getContext('2d');
    let particles = [];
    function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
    window.addEventListener('resize', resize); resize();
    class Particle {
        constructor(x, y, color) {
            this.x = x; this.y = y; this.color = color;
            this.velocity = { x: (Math.random() - 0.5) * 8, y: (Math.random() - 0.5) * 8 };
            this.alpha = 1; this.friction = 0.95;
        }
        draw() { ctx.save(); ctx.globalAlpha = this.alpha; ctx.beginPath(); ctx.arc(this.x, this.y, 2, 0, Math.PI * 2); ctx.fillStyle = this.color; ctx.fill(); ctx.restore(); }
        update() { this.velocity.x *= this.friction; this.velocity.y *= this.friction; this.x += this.velocity.x; this.y += this.velocity.y; this.alpha -= 0.01; }
    }
    function createFirework(x, y) {
        const colors = ['#ff6a00', '#ffae00', '#ffffff', '#ff4d4d'];
        for (let i = 0; i < 30; i++) particles.push(new Particle(x, y, colors[Math.floor(Math.random() * colors.length)]));
    }
    function animate() {
        requestAnimationFrame(animate);
        ctx.fillStyle = 'rgba(5, 7, 10, 0.15)'; ctx.fillRect(0, 0, canvas.width, canvas.height);
        if (Math.random() < 0.03) createFirework(Math.random() * canvas.width, Math.random() * canvas.height * 0.7);
        particles.forEach((p, i) => { if (p.alpha <= 0) particles.splice(i, 1); else { p.update(); p.draw(); } });
    }
    animate();

    if(sessionStorage.getItem('FILM_SESSION') === 'active') document.getElementById('authOverlay').style.display = 'none';
    renderUI();
